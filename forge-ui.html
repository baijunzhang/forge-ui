I'll split the monolithic authMiddleware into three focused functions that compose cleanly.

Here's the restructured approach — each layer handles a single concern and passes control via next():

typescript
// middleware/auth.ts export const validateJWT = async (req, res, next) => { const token = req.headers.authorization?.split(' ')[1] if (!token) return res.status(401).json({ error: 'No token' }) try { req.user = await verifyToken(token) next() } catch { res.status(401).json({ error: 'Invalid token' }) } } export const loadSession = async (req, res, next) => { req.session = await getSession(req.user.id) next() } export const requireRole = (role: string) => (req, res, next) => { if (!req.user.roles.includes(role)) return res.status(403).json({ error: 'Forbidden' }) next() }
